[documenti]    def create_dir(self, d):
        '''Create directory

        :param  str  - path directory

        '''
        dirname = d

        try:
            os.makedirs(dirname)
            return 1
        except OSError:
            if os.path.exists(dirname):
                # We are nearly safe
                return 0  # la cartella esiste
            else:
                # There was an error on creation, so make sure we know about it
                raise

[documenti]    def copy_file(self, f, d):
        """Copy file

        :param str: - file path

        :param str: - destination file

        """
        file_path = os.path.normpath(f)
        destination = os.path.normpath(d)
        if os.access(destination, 0):
            return 0  # la cartella esiste
        else:
            try:
                shutil.copy(file_path, destination)
                return 1
            except OSError:
                if os.path.exists(destination):
                    return 0
                else:
                    raise
